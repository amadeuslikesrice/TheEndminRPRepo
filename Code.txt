/* this shit is for volume control */
var audio = null;
var allowSelect = false;
var lastPlayAt = 0;

function ensureAudio() {
  if (!audio) { audio = new Audio(TEST_SOUND_URL); audio.preload = "auto"; }
  return audio;
}

function playTestSound(volume01) {
  var now = Date.now();
  if (now - lastPlayAt < 120) return;
  lastPlayAt = now;
  var a = ensureAudio();
  a.volume = Math.max(0, Math.min(1, volume01));
  try { a.currentTime = 0; } catch (e) {}
  a.play().then(function () {
    allowSelect = true;
    var btn = document.getElementById("btnSelect");
    if (btn) btn.disabled = false;
  }).catch(function () {});
}

function hideLoadingScreen(callback) {
  var screen = document.getElementById("LOADINGSCREEN");
  if (!screen) { if (callback) callback(); return; }
  screen.style.transition = "opacity 0.75s ease";
  screen.style.opacity = "0";
  setTimeout(function () { screen.remove(); if (callback) callback(); }, 1000);
}

function shortFileName(url) {
  try { var p = new URL(url, location.href).pathname; return decodeURIComponent(p.split("/").pop() || url); }
  catch (e) { return (url.split("/").pop() || url); }
}

function preloadOne(url) {
  return new Promise(function (resolve) {
    var img = new Image();
    img.crossOrigin = "anonymous";
    img.onload  = function () { resolve({ url: url, ok: true }); };
    img.onerror = function () { resolve({ url: url, ok: false }); };
    img.src = url;
  });
}

function setBarStatus(text) {
  var el = qs("#profileLoadbar .profileLoadbar-status");
  if (el) el.textContent = text || "";
}

function setBarProgress(pct) {
  var fill = qs("#profileLoadbar .profileLoadbar-fill");
  if (fill) fill.style.width = Math.max(0, Math.min(100, pct)) + "%";
}

async function runProfilePreload() {
  var bar = document.getElementById("profileLoadbar");
  if (bar) bar.style.opacity = "1";

  var seen = new Set();
  var list = [];
  for (var i = 0; i < PRELOAD_IMAGES.length; i++) {
    if (PRELOAD_IMAGES[i] && !seen.has(PRELOAD_IMAGES[i])) {
      seen.add(PRELOAD_IMAGES[i]); list.push(PRELOAD_IMAGES[i]);
    }
  }

  var total = list.length + 2;
  var done = 0;

  setBarStatus("// Initializing resources...");
  setBarProgress(0);

  /* preload transition audio in parallel */
  var transReady = false;
  transitionAudio = new Audio(SOUND_TRANSITION);
  transitionAudio.preload = "auto";
  var transPromise = new Promise(function (resolve) {
    transitionAudio.addEventListener("canplaythrough", function () {
      transReady = true;
      resolve();
    }, { once: true });
    setTimeout(function () { if (!transReady) resolve(); }, 15000);
  });

  var enterReady = false;
  enterLoginAudio = new Audio(SOUND_ENTER_LOGIN);
  enterLoginAudio.preload = "auto";
  var enterPromise = new Promise(function (resolve) {
    enterLoginAudio.addEventListener("canplaythrough", function () {
      enterReady = true;
      resolve();
    }, { once: true });
    setTimeout(function () { if (!enterReady) resolve(); }, 15000);
  });

  for (var j = 0; j < list.length; j++) {
    setBarStatus("// Loading: " + shortFileName(list[j]));
    await preloadOne(list[j]);
    done += 1;
    setBarProgress((done / total) * 45);
  }

  setBarStatus("// Loading: EndfieldTransistion.mp3");
  await transPromise;
  done += 1;
  setBarProgress((done / total) * 47);

  setBarStatus("// Loading: AK - Enter Login.mp3");
  await enterPromise;
  done += 1;
  setBarProgress((done / total) * 49);

  /* verify integrity */
  setBarStatus("// Verifying resource integrity...");
  setBarProgress(50);
  await new Promise(function (r) { setTimeout(r, 1500); });

  /* meme sequence */
  setBarStatus("// Pulling for SSR...");
  setBarProgress(60);
  await new Promise(function (r) { setTimeout(r, 2000); });

  setBarStatus("// Doing dailies...");
  setBarProgress(80);
  await new Promise(function (r) { setTimeout(r, 2000); });

  setBarStatus("// THEFACTORYMUSTGROW.EXE");
  setBarProgress(90);
  await new Promise(function (r) { setTimeout(r, 2000); });

  /* done */
  setBarStatus("// Complete.");
  setBarProgress(100);

  await new Promise(function (r) { setTimeout(r, 400); });

  if (bar) {
    bar.style.transition = "opacity 0.5s ease";
    bar.style.opacity = "0";
  }

  transitionAudio.volume = Math.min(1, (window.userVolume || 1) * 1.45);
  transitionAudio.currentTime = 0;
  transitionAudio.play().catch(function () {});

  runTransition();
}

      try { localStorage.setItem("user_volume", String(v)); } catch (e) {}

          cover.style.transition = "opacity 0.25s ease-in";

          setTimeout(function () { cover.remove(); }, 300);

        gate.setAttribute("aria-hidden", "false");

(function () {
  var MASKS = {
    body:   "https://i.imgur.com/RKU8bkn.png",
    head:   "https://i.imgur.com/ehaiTOz.png",
    tassle: "https://i.imgur.com/fwv6MSZ.png",
  };

  var PANELS = [
    "https://i.imgur.com/q7TCfAR.jpeg",
    "https://i.imgur.com/Ad3YqxT.jpeg",
    "https://i.imgur.com/uSLGIv5.jpeg",
  ];

  var root   = document.getElementById("Endmin");
  var canvas = document.getElementById("EndminCanvas");
  if (!root || !canvas) return;

  var rigBody   = root.querySelector(".EndminBody");
  var rigHead   = root.querySelector(".EndminHead");
  var rigTassle = root.querySelector(".EndminTassle");

  var ctx = canvas.getContext("2d", { alpha: true });

  var strip       = document.createElement("canvas");
  var stripCtx    = strip.getContext("2d", { alpha: true });
  var tmpPanel    = document.createElement("canvas");
  var tmpPanelCtx = tmpPanel.getContext("2d", { alpha: true });
  var part        = document.createElement("canvas");
  var partCtx     = part.getContext("2d", { alpha: true });

  var stripDirty = true;

  function loadImage(url) {
    return new Promise(function (resolve, reject) {
      var img = new Image();
      img.crossOrigin = "anonymous";
      img.onload  = function () { resolve(img); };
      img.onerror = reject;
      img.src = url;
    });
  }

  function setCanvasSize() {
    var r   = root.getBoundingClientRect();
    var dpr = Math.max(1, window.devicePixelRatio || 1);

    canvas.width  = Math.max(1, Math.floor(r.width * dpr));
    canvas.height = Math.max(1, Math.floor(r.height * dpr));
    canvas.style.width  = r.width + "px";
    canvas.style.height = r.height + "px";

    strip.height    = canvas.height;
    strip.width     = canvas.width * 12;
    tmpPanel.width  = canvas.width;
    tmpPanel.height = canvas.height;
    part.width      = canvas.width;
    part.height     = canvas.height;

    ctx.setTransform(1, 0, 0, 1, 0, 0);
    stripCtx.setTransform(1, 0, 0, 1, 0, 0);
    tmpPanelCtx.setTransform(1, 0, 0, 1, 0, 0);
    partCtx.setTransform(1, 0, 0, 1, 0, 0);

    stripDirty = true;
  }

  function drawCover(ctx2d, img, x, y, w, h) {
    var iw = img.naturalWidth || img.width;
    var ih = img.naturalHeight || img.height;
    var s  = Math.max(w / iw, h / ih);
    var dw = iw * s, dh = ih * s;
    ctx2d.drawImage(img, x + (w - dw) / 2, y + (h - dh) / 2, dw, dh);
  }

  function applyEdgeFade(ctx2d, w, h) {
    ctx2d.globalCompositeOperation = "destination-in";
    var g = ctx2d.createLinearGradient(0, 0, w, 0);
    g.addColorStop(0.00, "rgba(0,0,0,0)");
    g.addColorStop(0.10, "rgba(0,0,0,1)");
    g.addColorStop(0.90, "rgba(0,0,0,1)");
    g.addColorStop(1.00, "rgba(0,0,0,0)");
    ctx2d.fillStyle = g;
    ctx2d.fillRect(0, 0, w, h);
    ctx2d.globalCompositeOperation = "source-over";
  }

  function getMatrix(el) {
    var t = getComputedStyle(el).transform;
    if (!t || t === "none") return new DOMMatrix();
    return new DOMMatrix(t);
  }

  var speed = 75, hugRatio = 0.25, FPS = 30, frameInterval = 1000 / FPS;

  Promise.all(PANELS.map(loadImage))
    .then(function (loadedPanels) {
      return Promise.all([
        Promise.resolve(loadedPanels),
        loadImage(MASKS.body),
        loadImage(MASKS.head),
        loadImage(MASKS.tassle),
      ]);
    })
    .then(function (results) {
      var imgs       = results[0];
      var bodyMask   = results[1];
      var headMask   = results[2];
      var tassleMask = results[3];
      var masks = { bodyMask: bodyMask, headMask: headMask, tassleMask: tassleMask };

      setCanvasSize();
      window.addEventListener("resize", setCanvasSize);
      new ResizeObserver(function () { setCanvasSize(); }).observe(root);

      var x = 0, last = performance.now(), elapsed = 0;

      function buildStrip() {
        var cw = canvas.width, ch = canvas.height;
        var hug = cw * hugRatio, step = cw - hug;
        var needed = Math.ceil(strip.width / step) + 2;
        stripCtx.clearRect(0, 0, strip.width, strip.height);
        for (var i = 0; i < needed; i++) {
          tmpPanelCtx.clearRect(0, 0, cw, ch);
          drawCover(tmpPanelCtx, imgs[i % 3], 0, 0, cw, ch);
          applyEdgeFade(tmpPanelCtx, cw, ch);
          stripCtx.drawImage(tmpPanel, i * step, 0);
        }
        stripDirty = false;
      }

      function render(now) {
        if (window.endminStopped) return;
        var dt = now - last; last = now; elapsed += dt;
        if (elapsed < frameInterval) { requestAnimationFrame(render); return; }
        elapsed = 0;

        var dpr = Math.max(1, window.devicePixelRatio || 1);
        var cw = canvas.width, ch = canvas.height;
        var hug = cw * hugRatio, step = cw - hug;

        if (stripDirty) buildStrip();

        var loop = step * 3;
        x -= speed * (dt / 1000) * dpr;
        while (x <= -loop) x += loop;
        while (x > 0) x -= loop;

        ctx.clearRect(0, 0, cw, ch);

        function drawPart(maskImg, rigEl) {
          partCtx.clearRect(0, 0, cw, ch);
          partCtx.drawImage(strip, x, 0);
          partCtx.globalCompositeOperation = "destination-in";
          var m = getMatrix(rigEl);
          partCtx.setTransform(m.a, m.b, m.c, m.d, m.e * dpr, m.f * dpr);
          partCtx.drawImage(maskImg, 0, 0, cw, ch);
          partCtx.setTransform(1, 0, 0, 1, 0, 0);
          partCtx.globalCompositeOperation = "source-over";
          ctx.drawImage(part, 0, 0);
        }

        drawPart(masks.headMask, rigHead);
        drawPart(masks.tassleMask, rigTassle);
        drawPart(masks.bodyMask, rigBody);

        requestAnimationFrame(render);
      }

      requestAnimationFrame(render);
    })
    .catch(function (err) {
      console.error("Endmin canvas init failed:", err);
    });
})();

    var x = 0, last = performance.now(), elapsed = 0;

    function buildStrip() {
      var cw = walkCanvas.width, ch = walkCanvas.height;
      var step = cw - cw * hugRatio;
      var needed = Math.ceil(wStrip.width / step) + 2;
      wStripCtx.clearRect(0, 0, wStrip.width, wStrip.height);
      for (var i = 0; i < needed; i++) {
        wTmpCtx.clearRect(0, 0, cw, ch);
        drawCover(wTmpCtx, imgs[i % 3], 0, 0, cw, ch);
        applyEdgeFade(wTmpCtx, cw, ch);
        wStripCtx.drawImage(wTmp, i * step, 0);
      }
      wStripDirty = false;
    }

    function render(now) {
      if (!wRunning) return;
      var dt = now - last; last = now; elapsed += dt;
      if (elapsed < frameInterval) { requestAnimationFrame(render); return; }
      elapsed = 0;
      var dpr = Math.max(1, window.devicePixelRatio || 1);
      var cw = walkCanvas.width, ch = walkCanvas.height;
      var step = cw - cw * hugRatio;
      if (wStripDirty) buildStrip();
      var loop = step * 3;
      x -= speed * (dt / 1000) * dpr;
      while (x <= -loop) x += loop;
      while (x > 0) x -= loop;
      wCtx.clearRect(0, 0, cw, ch);
      wCtx.drawImage(wStrip, x, 0);
      requestAnimationFrame(render);
    }

    window.startEndminWalk = function () {
      wRunning = true;
      last = performance.now();
      elapsed = 0;
      requestAnimationFrame(render);
    };
  });
})();
